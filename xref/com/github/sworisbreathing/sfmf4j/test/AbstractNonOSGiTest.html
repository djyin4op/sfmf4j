28  /**
29   * Parent class for testing implementations in a typical J2SE environment.
30   *
31   * @author sswor
32   */
33  public abstract class AbstractNonOSGiTest<T extends FileMonitorServiceFactory> extends AbstractSFMF4JTest {
34  
35      /**
36       * Gets the implementation of {@link FileMonitorServiceFactory} being
37       * tested.
38       * @return the implementation being tested
39       */
40      protected abstract FileMonitorServiceFactory factoryInstance();
41  
42      /**
43       * The implementation of {@link FileMonitorServiceFactory} being tested.
44       */
45      private FileMonitorServiceFactory factoryInstance = null;
46  
47      /**
48       * The implementation of {@link FileMonitorService} being tested.
49       */
50      private FileMonitorService fileMonitor = null;
51  
52      /**
53       * Creates the {@link FileMonitorService} implementation being tested (from
54       * the {@link FileMonitorServiceFactory}) and initializes it, prior to
55       * testing.
56       */
57      @Before
58      public void setUp() {
59          factoryInstance = factoryInstance();
60          assert (factoryInstance != null);
61          fileMonitor = factoryInstance.createFileMonitorService();
62          assert (fileMonitor != null);
63          fileMonitor.initialize();
64      }
65  
66      /**
67       * Shuts down the {@link FileMonitorService} after testing.
68       */
69      @After
70      public void tearDown() {
71          if (fileMonitor != null) {
72              fileMonitor.shutdown();
73          }
74          fileMonitor = null;
75          factoryInstance = null;
76      }
77  
78      /**
79       * Runs the test against the file monitor implementation.
80       * @throws Throwable if the test fails
81       */
82      @Test
83      public void testFileMonitoring() throws Throwable {
84          doTestFileMonitoring(fileMonitor);
85      }
86      
87      /**
88       * Gets the implementation class being tested.
89       * @return the implementation class being tested
90       */
91      protected abstract Class<T> implementationClass();
92      
93      /**
94       * Tests {@link java.util.ServiceLoader} lookups. 
95       */
96      @Test
97      public void testSpi() {
98          ServiceLoader<FileMonitorServiceFactory> serviceLoader = ServiceLoader.load(FileMonitorServiceFactory.class);
99          assertNotNull(serviceLoader);
100         Iterator<FileMonitorServiceFactory> iterator = serviceLoader.iterator();
101         assertNotNull(iterator);
102         assertTrue(iterator.hasNext());
103         FileMonitorServiceFactory instance = iterator.next();
104         assertNotNull(instance);
105         assertTrue(implementationClass().isAssignableFrom(instance.getClass()));
106     }
